//How to bind the state data in jsx 
example:
export class App extends React.Component{

    constructor(){
      super();
      this.state={
        count:0
      }
    }

    render(){
       return (<div>
          <h1>Counter:{this.state.count}</h1>
          <button>ADD</button>
       </div>)
    }
}
